## 要补的题+题解
### J
一句话题解，按dfs序分块。
```c++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e5+10;
const int B = 300;
vector<int> G[N];
typedef long long ll;
int depth[N], clk, dfn[N], rr[N], vs[N];
unordered_map<int, int> bcnt[N/B+5];
ll btot[N/B+5], dtot[N];
void dfs(int v, int p = 0){   
    dfn[v] = clk;
    vs[clk] = v;
    bcnt[clk / B][depth[v]]++;
    ++clk;
    for (auto u: G[v]){
        if (u == p) continue;
        depth[u] = depth[v] + 1;
        dfs(u, v);
    }
    rr[v] = clk;
}
int get(int i, int d){
    return bcnt[i].count(d) ? bcnt[i][d] : 0;
}
int main() {
    int n,q;
    scanf("%d%d", &n, &q);
    for (int i = 0; i < n - 1; i++) {
        int u,v;
        scanf("%d%d", &u,&v);
        G[u].push_back(v);
        G[v].push_back(u);
    }
    dfs(1);
    while (q--) {
        int op;
        scanf("%d", &op);
        if (op == 1) {
            int d,x;
            scanf("%d%d", &d, &x);
            dtot[d] += x;
            for (int i = 0; i <= n / B; i++) {
                btot[i] += 1LL * x * get(i, d);
            }    
        }
        else {
            int v;
            scanf("%d", &v);
            int tl = dfn[v], tr = rr[v];
            ll ans = 0;
            while (tl < tr && tl % B != 0) ans += dtot[depth[vs[tl++]]];
            while (tl < tr && tr % B != 0) ans += dtot[depth[vs[--tr]]];
            while (tl < tr) {
                int b = tl / B;
                ans += btot[b];
                tl += B;
            }
            printf("%lld\n", ans);
        }
    }
}
```
